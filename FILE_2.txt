SET PAGESIZE 100
SET LINESIZE 100
SET COLSEP ' | '

-- Clear previous column formats (optional)
CLEAR COLUMNS;

-- Set formatting for output columns
COLUMN SupplierCompanyName FORMAT A35
COLUMN TotalQuotationAmount FORMAT 999,999,999

SELECT 
    sup.SupplierCompanyName,
    SUM(q.QuotationTotalAmount) AS TotalQuotationAmount
FROM 
    Quotation q
JOIN 
    Service s ON q.QuotationID = s.QuotationID
JOIN 
    Contract c ON s.ContractID = c.ContractID
JOIN 
    Supplier sup ON c.SupplierID = sup.SupplierID
WHERE 
    sup.SupplierStatus = 'Active'
    AND q.QuotationStatus = 'Approved'
GROUP BY 
    sup.SupplierCompanyName;
