-- Set up formatting for table-like output
SET LINESIZE 150
SET COLSEP ' | '
SET PAGESIZE 50

-- Define column formats and headings with updated attributes
COLUMN Employee_Email HEADING 'Employee Email' FORMAT A25
COLUMN WorkOrder_Priority HEADING 'Work Order Priority' FORMAT A10
COLUMN Service_RefNo HEADING 'Service Ref No' FORMAT A11
COLUMN Contract_Status HEADING 'Contract Status' FORMAT A15
COLUMN Setup_Status HEADING 'Setup Status' FORMAT A12
COLUMN Report_Code HEADING 'Report Code' FORMAT A12
COLUMN Feedback_Rating HEADING 'Rating' FORMAT 99
COLUMN Hardware_Support_Type HEADING 'Hardware Support Type' FORMAT A21

-- Query to join eight tables with updated SELECT attributes
SELECT 
    E.EmployeeWorkEmail AS Employee_Email,
    W.WorkOrderPriority AS WorkOrder_Priority,
    S.ServiceRefNo AS Service_RefNo,
    C.ContractStatus AS Contract_Status,
    ST.SetupStatus AS Setup_Status,
    SR.ReportCode AS Report_Code,
    CF.FeedbackRating AS Feedback_Rating,
    H.SupportType AS Hardware_Support_Type
FROM 
    Employee E
    JOIN WorkOrder W ON E.EmployeeID = W.EmployeeID
    JOIN Service_WorkOrder SW ON W.WorkOrderID = SW.WorkOrderID
    JOIN Service S ON SW.ServiceID = S.ServiceID
    JOIN Contract C ON S.ContractID = C.ContractID
    JOIN Service_Setup SS ON S.ServiceID = SS.ServiceID
    JOIN Setup ST ON SS.SetupID = ST.SetupID
    JOIN ServiceReport SR ON ST.SetupID = SR.SetupID
    JOIN ClientFeedback CF ON SR.ReportID = CF.ReportID
    JOIN HardwareSupp H ON S.ServiceID = H.ServiceID
WHERE 
    E.EmployeeStatus = 'Active'
    AND W.WorkOrderCreatedDate > TO_DATE('2023-01-01', 'YYYY-MM-DD')
    AND S.ServiceStartDate > TO_DATE('2023-01-01', 'YYYY-MM-DD')
    AND CF.FeedbackRating >= 1
ORDER BY 
    CF.FeedbackRating DESC;
