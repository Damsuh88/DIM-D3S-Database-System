-- 1. Person
CREATE TABLE Person (
    PersonID NUMBER(10) NOT NULL UNIQUE, -- Part of composite PK, also UNIQUE to support single-column FKs
    PersonRefNo VARCHAR2(20) UNIQUE NOT NULL, -- Part of composite PK, also UNIQUE
    ICNo VARCHAR2(12) UNIQUE NOT NULL,
    PersonStatus VARCHAR2(20) DEFAULT 'Pending' NOT NULL CHECK (PersonStatus IN ('Active','Inactive','Pending')),
    CONSTRAINT pk_person PRIMARY KEY (PersonID, PersonRefNo) -- Composite Primary Key
);

-- 2. Customer 
CREATE TABLE Customer (
    CustomerID NUMBER(10) NOT NULL UNIQUE,
    CustomerCode VARCHAR2(20) UNIQUE NOT NULL,
    CustomerCompanyName VARCHAR2(100) NOT NULL,
    CustomerContactPerson VARCHAR2(100) NOT NULL,
    CustomerIndustry VARCHAR2(50) NOT NULL,
